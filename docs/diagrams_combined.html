Database Relationships

  1. User to Restaurant (1-to-many)
    • One user can own many restaurants (owner_id)
    • Each restaurant belongs to one user
  2. City to Restaurant (1-to-many)
    • One city can have many restaurants
    • Each restaurant belongs to one city

Table Structure

Users Table

CREATE TABLE users (
    id bigint unsigned NOT NULL AUTO_INCREMENT,
    name varchar(255) NOT NULL,
    email varchar(255) NOT NULL UNIQUE,
    email_verified_at timestamp NULL,
    password varchar(255) NOT NULL,
    remember_token varchar(100) NULL,
    created_at timestamp NULL,
    updated_at timestamp NULL,
    PRIMARY KEY (id)
);

Cities Table

CREATE TABLE cities (
    id bigint unsigned NOT NULL AUTO_INCREMENT,
    name varchar(255) NOT NULL,
    created_at timestamp NULL,
    updated_at timestamp NULL,
    PRIMARY KEY (id)
);

Restaurants Table

CREATE TABLE restaurants (
    id bigint unsigned NOT NULL AUTO_INCREMENT,
    owner_id bigint unsigned NOT NULL,
    city_id bigint unsigned NOT NULL,
    name varchar(255) NOT NULL,
    address text NOT NULL,
    created_at timestamp NULL,
    updated_at timestamp NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (owner_id) REFERENCES users(id),
    FOREIGN KEY (city_id) REFERENCES cities(id)
);
